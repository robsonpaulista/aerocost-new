rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Bloquear acesso direto do cliente
    // A aplicação usa API Routes (server-side) para acessar o Firestore
    // Isso é mais seguro porque as validações são feitas no servidor
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
