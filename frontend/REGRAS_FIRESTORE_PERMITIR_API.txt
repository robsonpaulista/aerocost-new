rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Permitir leitura/escrita para todas as coleções
    // As validações são feitas no servidor (API Routes do Next.js)
    // Isso é seguro porque:
    // 1. As API Routes rodam no servidor
    // 2. Todas as operações passam por validação de autenticação/autorização
    // 3. Não há acesso direto do cliente ao Firestore
    match /{document=**} {
      allow read, write: if true;
    }
  }
}
